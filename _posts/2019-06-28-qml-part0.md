# PyQGIS, QML and charts


<br/>
```python
from PyQt5 import QtCore

class LineSeries(QtCore.QObject):

  def __init__(self):
    self.init_gui()

  def init_gui(self):
    self._view = QtQuick.QQuickView(self.parent)
    self._view.setResizeMode(QtQuick.QQuickView.SizeRootObjectToView)
    self._view.rootContext().setContextProperty("bazinga", self)

    qml = "/home/pblottiere/devel/perso/bazinga/qml/lines.qml"
    self._view.setSource(QtCore.QUrl.fromLocalFile(qml))

    self.container = QtWidgets.QWidget.createWindowContainer(self._view)
    self._widget.layout().addWidget(self.container)
````
